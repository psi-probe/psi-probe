108   /**
109    * Builds the delta stats.
110    *
111    * @param name the name
112    * @param value the value
113    * @return the long
114    * @throws InterruptedException the interrupted exception
115    */
116   protected long buildDeltaStats(String name, long value) throws InterruptedException {
117     return buildDeltaStats(name, value, System.currentTimeMillis());
118   }
119 
120   /**
121    * Builds the delta stats.
122    *
123    * @param name the name
124    * @param value the value
125    * @param time the time
126    * @return the long
127    * @throws InterruptedException the interrupted exception
128    */
129   protected long buildDeltaStats(String name, long value, long time) throws InterruptedException {
130     long delta = 0;
131     if (statsCollection != null) {
132       long previousValue = Utils.toLong(previousData.get(name), 0);
133       delta = value - previousValue;
134       delta = delta > 0 ? delta : 0;
135       buildAbsoluteStats(name, delta, time);
136       previousData.put(name, value);
137     }
138     return delta;
139   }
140 
141   /**
142    * Builds the absolute stats.
143    *
144    * @param name the name
145    * @param value the value
146    * @throws InterruptedException the interrupted exception
147    */
148   protected void buildAbsoluteStats(String name, long value) throws InterruptedException {
149     buildAbsoluteStats(name, value, System.currentTimeMillis());
150   }
151 
152   /**
153    * Builds the absolute stats.
154    *
155    * @param name the name
156    * @param value the value
157    * @param time the time
158    * @throws InterruptedException the interrupted exception
159    */
160   protected void buildAbsoluteStats(String name, long value, long time)
161       throws InterruptedException {
162 
163     List<XYDataItem> stats = statsCollection.getStats(name);
164     if (stats == null) {
165       statsCollection.newStats(name, maxSeries);
166     } else {
167       XYDataItem data = new XYDataItem(time, value);
168       statsCollection.lockForUpdate();
169       try {
170         stats.add(data);
171         houseKeepStats(stats);
172       } finally {
173         statsCollection.releaseLock();
174       }
175       if (listeners != null) {
176         StatsCollectionEvent event = new StatsCollectionEvent(name, data);
177         for (StatsCollectionListener listener : listeners) {
178           if (listener.isEnabled()) {
179             listener.statsCollected(event);
180           }
181         }
182       }
183     }
184   }

206   /**
207    * If there is a value indicating the accumulated amount of time spent on something it is possible
208    * to build a series of values representing the percentage of time spent on doing something. For
209    * example:
210    *
211    * <p>
212    * at point T1 the system has spent A milliseconds performing tasks at point T2 the system has
213    * spent B milliseconds performing tasks
214    * </p>
215    *
216    * <p>
217    * so between in a timeframe T2-T1 the system spent B-A milliseconds being busy. Thus (B - A)/(T2
218    * - T1) * 100 is the percentage of all time the system spent being busy.
219    * </p>
220    *
221    * @param name the name of the value, which allows the collector to be used for multiple values
222    * @param value time spent on the task in milliseconds (A or B in the example above)
223    * @param time system time in milliseconds (T1 or T2 in the example above)
224    * @throws InterruptedException if a lock could not be obtained
225    */
226   protected void buildTimePercentageStats(String name, long value, long time)
227       throws InterruptedException {
228 
229     Entry entry = previousData2D.get(name);
230     if (entry == null) {
231       entry = new Entry();
232       entry.value = value;
233       entry.time = time;
234       previousData2D.put(name, entry);
235     } else {
236       double valueDelta = (double) value - entry.value;
237       double timeDelta = (double) time - entry.time;
238       double statValue = valueDelta * 100 / timeDelta;
239       statsCollection.lockForUpdate();
240       try {
241         List<XYDataItem> stats = statsCollection.getStats(name);
242         if (stats == null) {
243           stats = statsCollection.newStats(name, maxSeries);
244         }
245         stats.add(stats.size(), new XYDataItem(time, statValue));
246         houseKeepStats(stats);
247       } finally {
248         statsCollection.releaseLock();
249       }
250     }
251   }

253   /**
254    * Reset stats.
255    *
256    * @param name the name
257    */
258   protected void resetStats(String name) {
259     statsCollection.resetStats(name);
260   }

262   /**
263    * House keep stats.
264    *
265    * @param stats the stats
266    */
267   private void houseKeepStats(List<XYDataItem> stats) {
268     while (stats.size() > maxSeries) {
269       stats.remove(0);
270     }
271   }
