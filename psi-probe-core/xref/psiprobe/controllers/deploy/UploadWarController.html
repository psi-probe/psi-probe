18  
19  import javax.servlet.http.HttpServletRequest;
20  import javax.servlet.http.HttpServletResponse;
21  
22  import org.apache.catalina.Context;
23  import org.apache.commons.io.FileUtils;
24  import org.apache.commons.io.FilenameUtils;
25  import org.apache.tomcat.util.http.fileupload.FileItem;
26  import org.apache.tomcat.util.http.fileupload.FileItemFactory;
27  import org.apache.tomcat.util.http.fileupload.disk.DiskFileItemFactory;
28  import org.apache.tomcat.util.http.fileupload.servlet.ServletFileUpload;
29  import org.apache.tomcat.util.http.fileupload.servlet.ServletRequestContext;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  import org.springframework.beans.factory.annotation.Value;
33  import org.springframework.security.core.Authentication;
34  import org.springframework.security.core.context.SecurityContextHolder;
35  import org.springframework.stereotype.Controller;
36  import org.springframework.web.bind.annotation.RequestMapping;
37  import org.springframework.web.servlet.ModelAndView;
38  import org.springframework.web.servlet.view.InternalResourceView;

41  import psiprobe.controllers.jsp.DisplayJspController;

44  /**
45   * Uploads and installs web application from a .WAR.
46   */
47  @Controller
48  public class UploadWarController extends AbstractTomcatContainerController {
49  
50    /** The Constant logger. */
51    private static final Logger logger = LoggerFactory.getLogger(UploadWarController.class);
52    private static final int MAXSECONDS_WAITFOR_CONTEXT = 10;
53  
54    @RequestMapping(path = "/adm/war.htm")
55    @Override
56    public ModelAndView handleRequest(HttpServletRequest request, HttpServletResponse response)
57        throws Exception {
58      return super.handleRequest(request, response);
59    }
60  
61    @Override
62    protected ModelAndView handleRequestInternal(HttpServletRequest request,
63        HttpServletResponse response) throws Exception {
64  
65      // If not multi-part content, exit
66      if (!ServletFileUpload.isMultipartContent(request)) {
67        return new ModelAndView(new InternalResourceView(getViewName()));
68      }
69  
70      File tmpWar = null;
71      String contextName = null;
72      boolean update = false;
73      boolean compile = false;
74      boolean discard = false;
75  
76      // parse multipart request and extract the file
77      FileItemFactory factory =
78          new DiskFileItemFactory(1048000, new File(System.getProperty("java.io.tmpdir")));
79      ServletFileUpload upload = new ServletFileUpload(factory);
80      upload.setSizeMax(-1);
81      upload.setHeaderEncoding(StandardCharsets.UTF_8.name());
82      try {
83        List<FileItem> fileItems = upload.parseRequest(new ServletRequestContext(request));
84        for (FileItem fi : fileItems) {
85          if (!fi.isFormField()) {
86            if (fi.getName() != null && fi.getName().length() > 0) {
87              tmpWar =
88                  new File(System.getProperty("java.io.tmpdir"), FilenameUtils.getName(fi.getName()));
89              fi.write(tmpWar);
90            }
91          } else if ("context".equals(fi.getFieldName())) {
92            contextName = fi.getString();
93          } else if ("update".equals(fi.getFieldName()) && "yes".equals(fi.getString())) {
94            update = true;
95          } else if ("compile".equals(fi.getFieldName()) && "yes".equals(fi.getString())) {
96            compile = true;
97          } else if ("discard".equals(fi.getFieldName()) && "yes".equals(fi.getString())) {
98            discard = true;
99          }
100       }
101     } catch (Exception e) {
102       logger.error("Could not process file upload", e);
103       request.setAttribute("errorMessage", getMessageSourceAccessor()
104           .getMessage("probe.src.deploy.war.uploadfailure", new Object[] {e.getMessage()}));
105       if (tmpWar != null) {
106         Files.delete(tmpWar.toPath());
107         tmpWar = null;
108       }
109     }
110 
111     String errMsg = null;
112 
113     if (tmpWar == null) {
114       return new ModelAndView(new InternalResourceView(getViewName()));
115     }
116 
117     try {
118       if (tmpWar.getName().endsWith(".war")) {
119 
120         if (contextName == null || contextName.length() == 0) {
121           String warFileName = tmpWar.getName().replaceAll("\\.war$", "");
122           contextName = "/" + warFileName;
123         }
124 
125         contextName = getContainerWrapper().getTomcatContainer().formatContextName(contextName);
126 
127         /*
128          * pass the name of the newly deployed context to the presentation layer using this name the
129          * presentation layer can render a url to view compilation details
130          */
131         String visibleContextName = contextName.isEmpty() ? "/" : contextName;
132         request.setAttribute("contextName", visibleContextName);
133 
134         if (update && getContainerWrapper().getTomcatContainer().findContext(contextName) != null) {
135           logger.debug("updating {}: removing the old copy", contextName);
136           getContainerWrapper().getTomcatContainer().remove(contextName);
137         }
138 
139         if (getContainerWrapper().getTomcatContainer().findContext(contextName) == null) {
140           // move the .war to tomcat application base dir
141           String destWarFilename =
142               getContainerWrapper().getTomcatContainer().formatContextFilename(contextName);
143           File destWar = new File(getContainerWrapper().getTomcatContainer().getAppBase(),
144               destWarFilename + ".war");
145 
146           FileUtils.moveFile(tmpWar, destWar);
147 
148           // let Tomcat know that the file is there
149           getContainerWrapper().getTomcatContainer().installWar(contextName);
150 
151           File destContext =
152               new File(getContainerWrapper().getTomcatContainer().getAppBase(), destWarFilename);
153           // Wait few seconds for creating context dir to avoid empty context
154           FileUtils.waitFor(destContext, MAXSECONDS_WAITFOR_CONTEXT);
155 
156           Context ctx = getContainerWrapper().getTomcatContainer().findContext(contextName);
157           if (ctx == null) {
158             errMsg = getMessageSourceAccessor().getMessage("probe.src.deploy.war.notinstalled",
159                 new Object[] {visibleContextName});
160           } else {
161             request.setAttribute("success", Boolean.TRUE);
162             // Logging action
163             Authentication auth = SecurityContextHolder.getContext().getAuthentication();
164             // get username logger
165             String name = auth.getName();
166             logger.info(getMessageSourceAccessor().getMessage("probe.src.log.deploywar"), name,
167                 contextName);
168             if (discard) {
169               getContainerWrapper().getTomcatContainer().discardWorkDir(ctx);
170               logger.info(getMessageSourceAccessor().getMessage("probe.src.log.discardwork"), name,
171                   contextName);
172             }
173             if (compile) {
174               Summary summary = new Summary();
175               summary.setName(ctx.getName());
176               getContainerWrapper().getTomcatContainer().listContextJsps(ctx, summary, true);
177               request.getSession(false).setAttribute(DisplayJspController.SUMMARY_ATTRIBUTE,
178                   summary);
179               request.setAttribute("compileSuccess", Boolean.TRUE);
180             }
181           }
182         } else {
183           errMsg = getMessageSourceAccessor().getMessage("probe.src.deploy.war.alreadyExists",
184               new Object[] {visibleContextName});
185         }
186       } else {
187         errMsg = getMessageSourceAccessor().getMessage("probe.src.deploy.war.notWar.failure");
188       }
189     } catch (IOException e) {
190       errMsg = getMessageSourceAccessor().getMessage("probe.src.deploy.war.failure",
191           new Object[] {e.getMessage()});
192       logger.error("Tomcat threw an exception when trying to deploy", e);
193     } finally {
194       if (errMsg != null) {
195         request.setAttribute("errorMessage", errMsg);
196       }
197       // If war was not moved, delete it
198       if (tmpWar.exists()) {
199         Files.delete(tmpWar.toPath());
200       }
201     }
202     return new ModelAndView(new InternalResourceView(getViewName()));
203   }
204 
205   @Value("/adm/deploy.htm")
206   @Override
207   public void setViewName(String viewName) {
208     super.setViewName(viewName);
209   }
210 
211 }
