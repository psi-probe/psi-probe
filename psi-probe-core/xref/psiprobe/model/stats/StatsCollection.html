175   /**
176    * Gets the last value for stat.
177    *
178    * @param statName the stat name
179    * @return the last value for stat
180    */
181   public long getLastValueForStat(String statName) {
182     long statValue = 0;
183 
184     List<XYDataItem> stats = getStats(statName);
185     if (stats != null && !stats.isEmpty()) {
186       XYDataItem xy = stats.get(stats.size() - 1);
187       if (xy != null && xy.getY() != null) {
188         statValue = xy.getY().longValue();
189       }
190     }
191 
192     return statValue;
193   }

195   /**
196    * Returns series if stat name starts with the prefix.
197    *
198    * @param statNamePrefix they key under which the stats are stored
199    * @return a Map of matching stats. Map keys are stat names and map values are corresponding
200    *         series.
201    */
202   public synchronized Map<String, List<XYDataItem>> getStatsByPrefix(String statNamePrefix) {
203     Map<String, List<XYDataItem>> map = new HashMap<>();
204     for (Map.Entry<String, List<XYDataItem>> en : statsData.entrySet()) {
205       if (en.getKey().startsWith(statNamePrefix)) {
206         map.put(en.getKey(), en.getValue());
207       }
208     }
209     return map;
210   }

212   /**
213    * Make file.
214    *
215    * @return the file
216    */
217   private File makeFile() {
218     return storagePath == null ? new File(contextTempDir, swapFileName)
219         : new File(storagePath, swapFileName);
220   }
221 
222   /**
223    * Shift files.
224    *
225    * @param index the index
226    */
227   private void shiftFiles(int index) {
228     if (index >= maxFiles - 1) {
229       try {
230         if (Files.exists(Paths.get(makeFile().getAbsolutePath() + "." + index))) {
231           Files.delete(Paths.get(makeFile().getAbsolutePath() + "." + index));
232         }
233       } catch (IOException e) {
234         logger.error("Could not delete file {}",
235             new File(makeFile().getAbsolutePath() + "." + index).getName());
236       }
237     } else {
238       shiftFiles(index + 1);
239       File srcFile = index == 0 ? makeFile() : new File(makeFile().getAbsolutePath() + "." + index);
240       if (Files.exists(srcFile.toPath())) {
241         File destFile = new File(makeFile().getAbsolutePath() + "." + (index + 1));
242         if (!srcFile.renameTo(destFile)) {
243           logger.error("Could not rename file {} to {}", srcFile.getName(), destFile.getName());
244         }
245       }
246     }
247   }

249   /**
250    * Writes stats data to file on disk.
251    *
252    * @throws InterruptedException if a lock cannot be obtained
253    */
254   public synchronized void serialize() throws InterruptedException {
255     lock.lockForCommit();
256     long start = System.currentTimeMillis();
257     try {
258       shiftFiles(0);
259       try (OutputStream os = Files.newOutputStream(makeFile().toPath())) {
260         xstream.toXML(statsData, os);
261       }
262     } catch (Exception e) {
263       logger.error("Could not write stats data to '{}'", makeFile().getAbsolutePath(), e);
264     } finally {
265       lock.releaseCommitLock();
266       logger.debug("stats serialized in {}ms", System.currentTimeMillis() - start);
267     }
268   }

270   /**
271    * Deserialize.
272    *
273    * @param file the file
274    * @return the map
275    */
276   @SuppressWarnings("unchecked")
277   private Map<String, List<XYDataItem>> deserialize(File file) {
278     Map<String, List<XYDataItem>> stats = null;
279     if (file.exists() && file.canRead()) {
280       long start = System.currentTimeMillis();
281       try {
282         try (InputStream fis = Files.newInputStream(file.toPath())) {
283           stats = (Map<String, List<XYDataItem>>) (xstream.fromXML(fis));
284 
285           if (stats != null) {
286             // adjust stats data so that charts look realistic.
287             // we do that by ending the previous stats group with 0 value
288             // and starting the current stats group also with 0
289             // thus giving the chart nice plunge to zero indicating downtime
290             // and lets not bother about rotating stats;
291             // regular stats collection cycle will do it
292 
293             for (Entry<String, List<XYDataItem>> set : stats.entrySet()) {
294               List<XYDataItem> list = set.getValue();
295               if (!list.isEmpty()) {
296                 XYDataItem xy = list.get(list.size() - 1);
297                 list.add(new XYDataItem(xy.getX().longValue() + 1, 0));
298                 list.add(new XYDataItem(System.currentTimeMillis(), 0));
299               }
300             }
301           }
302         }
303         logger.debug("stats data read in {}ms", System.currentTimeMillis() - start);
304       } catch (Exception e) {
305         logger.error("Could not read stats data from '{}'", file.getAbsolutePath(), e);
306       }
307     }
308 
309     return stats;
310   }

328   /**
329    * Reads stats data from file on disk.
330    */
331   @Override
332   public synchronized void afterPropertiesSet() {
333     int index = 0;
334     Map<String, List<XYDataItem>> stats;
335 
336     while (true) {
337       File file = index == 0 ? makeFile() : new File(makeFile().getAbsolutePath() + "." + index);
338       stats = deserialize(file);
339       index += 1;
340       if (stats != null || index >= maxFiles - 1) {
341         break;
342       }
343     }
344 
345     if (stats != null) {
346       statsData = stats;
347     } else {
348       logger.debug("Stats data file not found. Empty file assumed.");
349     }
350 
351   }

354   public void destroy() throws Exception {
355     serialize();
356   }

362   }
