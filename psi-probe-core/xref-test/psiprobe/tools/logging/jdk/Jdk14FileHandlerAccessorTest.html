11  package psiprobe.tools.logging.jdk;
12  
13  import java.io.File;
14  import java.io.IOException;
15  import java.nio.file.Files;
16  import java.nio.file.Path;
17  import java.util.logging.FileHandler;
18  
19  import org.assertj.core.api.Assertions;
20  import org.junit.jupiter.api.Test;
21  
22  import psiprobe.model.Application;
23  
24  /**
25   * The Class Jdk14FileHandlerAccessorTest.
26   */
27  class Jdk14FileHandlerAccessorTest {
28  
29    /**
30     * Gets the file.
31     *
32     * @return the file
33     * @throws IOException Signals that an I/O exception has occurred.
34     */
35    @Test
36    void getFile() throws IOException {
37      Jdk14FileHandlerAccessor handlerAccessor = new Jdk14FileHandlerAccessor();
38      handlerAccessor.setLoggerAccessor(new Jdk14LoggerAccessor());
39  
40      Path testPath = Files.createTempDirectory("psi-probe");
41      testPath.toFile().deleteOnExit();
42      FileHandler target = new FileHandler(testPath.toString() + "test-%g.log", 1024, 3);
43  
44      handlerAccessor.setTarget(target);
45      handlerAccessor.setIndex(Integer.toString(0));
46      Application testApplication = new Application();
47      handlerAccessor.setApplication(testApplication);
48  
49      File file = handlerAccessor.getFile();
50  
51      Assertions.assertThat(file.getAbsolutePath()).isEqualTo(testPath + "test-0.log");
52  
53    }
54  
55  }
