11  package psiprobe.tools.logging.slf4jlogback;
12  
13  import java.lang.reflect.InvocationTargetException;
14  import java.lang.reflect.Method;
15  import java.util.ArrayList;
16  import java.util.List;
17  
18  import org.apache.commons.lang3.reflect.MethodUtils;
19  
20  import psiprobe.tools.logging.DefaultAccessor;
21  
22  /**
23   * Wraps a TomcatSlf4jLogback logger factory from a given web application class loader.
24   * 
25   * <p>
26   * All TomcatSlf4jLogback classes are loaded via the given class loader and not via psi-probe's own
27   * class loader. For this reasons, all methods on TomcatSlf4jLogback objects are invoked via
28   * reflection.
29   * </p>
30   * <p>
31   * This way, we can even handle different versions of TomcatSlf4jLogback embedded in different WARs.
32   * </p>
33   */
34  public class TomcatSlf4jLogbackFactoryAccessor extends DefaultAccessor {
35  
36    /**
37     * Attempts to initialize a TomcatSlf4jLogback logger factory via the given class loader.
38     *
39     * @param cl the ClassLoader to use when fetching the factory
40     * @throws ClassNotFoundException the class not found exception
41     * @throws IllegalAccessException the illegal access exception
42     * @throws InvocationTargetException the invocation target exception
43     */
44    public TomcatSlf4jLogbackFactoryAccessor(ClassLoader cl)
45        throws ClassNotFoundException, IllegalAccessException, InvocationTargetException {
46  
47      // Get the singleton SLF4J binding, which may or may not be Logback, depending on the binding.
48      Class<?> clazz = cl.loadClass("org.apache.juli.logging.org.slf4j.impl.StaticLoggerBinder");
49      Method getSingleton = MethodUtils.getAccessibleMethod(clazz, "getSingleton");
50      Object singleton = getSingleton.invoke(null);
51      Method getLoggerFactory = MethodUtils.getAccessibleMethod(clazz, "getLoggerFactory");
52  
53      Object loggerFactory = getLoggerFactory.invoke(singleton);
54  
55      // Check if the binding is indeed Logback
56      Class<?> loggerFactoryClass =
57          cl.loadClass("org.apache.juli.logging.ch.qos.logback.classic.LoggerContext");
58      if (!loggerFactoryClass.isInstance(loggerFactory)) {
59        throw new RuntimeException("The singleton SLF4J binding was not Logback");
60      }
61      setTarget(loggerFactory);
62    }
63  
64    /**
65     * Returns the TomcatSlf4jLogback root logger.
66     *
67     * @return the root logger
68     */
69    public TomcatSlf4jLogbackLoggerAccessor getRootLogger() {
70      /*
71       * TomcatSlf4jLogback has no dedicated getRootLogger() method, so we simply access the root
72       * logger by its well-defined name.
73       */
74      return getLogger("ROOT");
75    }
76  
77    /**
78     * Returns the TomcatSlf4jLogback logger with a given name.
79     *
80     * @param name the name
81     * @return the Logger with the given name
82     */
83    public TomcatSlf4jLogbackLoggerAccessor getLogger(String name) {
84      try {
85        Class<? extends Object> clazz = getTarget().getClass();
86        Method getLogger = MethodUtils.getAccessibleMethod(clazz, "getLogger", String.class);
87  
88        Object logger = getLogger.invoke(getTarget(), name);
89        if (logger == null) {
90          throw new NullPointerException(getTarget() + ".getLogger(\"" + name + "\") returned null");
91        }
92        TomcatSlf4jLogbackLoggerAccessor accessor = new TomcatSlf4jLogbackLoggerAccessor();
93        accessor.setTarget(logger);
94        accessor.setApplication(getApplication());
95        return accessor;
96  
97      } catch (Exception e) {
98        logger.error("{}.getLogger('{}') failed", getTarget(), name, e);
99      }
100     return null;
101   }
102 
103   /**
104    * Returns a list of wrappers for all TomcatSlf4jLogback appenders that have an associated logger.
105    *
106    * @return a list of {@link TomcatSlf4jLogbackAppenderAccessor}s representing all appenders that
107    *         are in use
108    */
109   @SuppressWarnings("unchecked")
110   public List<TomcatSlf4jLogbackAppenderAccessor> getAppenders() {
111     List<TomcatSlf4jLogbackAppenderAccessor> appenders = new ArrayList<>();
112     try {
113       Class<? extends Object> clazz = getTarget().getClass();
114       Method getLoggerList = MethodUtils.getAccessibleMethod(clazz, "getLoggerList");
115 
116       List<Object> loggers = (List<Object>) getLoggerList.invoke(getTarget());
117       for (Object logger : loggers) {
118         TomcatSlf4jLogbackLoggerAccessor accessor = new TomcatSlf4jLogbackLoggerAccessor();
119         accessor.setTarget(logger);
120         accessor.setApplication(getApplication());
121 
122         appenders.addAll(accessor.getAppenders());
123       }
124     } catch (Exception e) {
125       logger.error("{}.getLoggerList() failed", getTarget(), e);
126     }
127     return appenders;
128   }
