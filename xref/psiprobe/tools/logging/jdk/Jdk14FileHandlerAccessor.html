11  package psiprobe.tools.logging.jdk;
12  
13  import java.io.File;
14  
15  import psiprobe.tools.Instruments;
16  
17  /**
18   * JDK 1.4 Logging File Handler Accessor Class.
19   */
20  public class Jdk14FileHandlerAccessor extends Jdk14HandlerAccessor {
21  
22    private static final int LATEST_FILE_INDEX = 0;
23  
24    /**
25     * Currently, we only access the latest log file with index 0.
26     */
27    @Override
28    public File getFile() {
29      File[] files = (File[]) Instruments.getField(getTarget(), "files");
30      if (files == null || files.length == 0) {
31        throw new IllegalStateException("File handler does not manage any files");
32      }
33      return files[LATEST_FILE_INDEX];
34    }
35  
36  }
