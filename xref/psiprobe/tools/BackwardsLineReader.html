11  package psiprobe.tools;
12  
13  import java.io.BufferedInputStream;
14  import java.io.ByteArrayOutputStream;
15  import java.io.IOException;
16  import java.io.InputStream;
17  import java.nio.charset.Charset;
18  import java.nio.charset.StandardCharsets;
19  
20  /**
21   * Reads lines from "backwards" InputStream. This class facilitates reading files from bottom up.
22   *
23   * <p>
24   * This source code was kindly contributed by Kan Ogawa.
25   * </p>
26   */
27  public class BackwardsLineReader {
28  
29    /** The bis. */
30    private BufferedInputStream bis;
31  
32    /** The skip line feed. */
33    private boolean skipLineFeed = true;
34  
35    /** The encoding. */
36    private String encoding;
37  
38    /**
39     * Instantiates a new backwards line reader.
40     *
41     * @param is the is
42     */
43    public BackwardsLineReader(InputStream is) {
44      this(is, null);
45    }
46  
47    /**
48     * Instantiates a new backwards line reader.
49     *
50     * @param is the is
51     * @param encoding the encoding
52     */
53    public BackwardsLineReader(InputStream is, String encoding) {
54      this.bis = new BufferedInputStream(is, 8192);
55      this.encoding = encoding;
56    }
57  
58    /**
59     * Read line.
60     *
61     * @return the string
62     * @throws IOException Signals that an I/O exception has occurred.
63     */
64    public String readLine() throws IOException {
65      ByteArrayOutputStream baos = new ByteArrayOutputStream(512);
66      boolean empty = false;
67      while (true) {
68        byte chr = (byte) bis.read();
69        if (chr == -1) {
70          // quit this loop, if the first of the backwards stream is
71          // reached
72          if (baos.toByteArray().length == 0) {
73            empty = true;
74          }
75          break;
76        }
77        if (chr == '\n') {
78          skipLineFeed = false;
79          // quit this loop
80          break;
81        }
82        if (chr == '\r') {
83          if (skipLineFeed) {
84            // quit this loop. if the carriage return only was read
85            break;
86          }
87          // go to next loop, if both the carriage return and
88          // the line feed were read
89          continue;
90        }
91        baos.write(chr);
92      }
93      if (!empty) {
94        byte[] byteArray = baos.toByteArray();
95        reverse(byteArray);
96        return encoding == null ? new String(byteArray, StandardCharsets.UTF_8)
97            : new String(byteArray, Charset.forName(encoding));
98      }
99      // return null if the end of the stream has been reached
100     return null;
101   }
102 
103   /**
104    * Close.
105    *
106    * @throws IOException Signals that an I/O exception has occurred.
107    */
108   public void close() throws IOException {
109     if (bis != null) {
110       bis.close();
111     }
112   }
113 
114   /**
115    * Reverse.
116    *
117    * @param byteArray the byte array
118    */
119   private void reverse(byte[] byteArray) {
120     for (int i = 0; i < byteArray.length / 2; i++) {
121       byte temp = byteArray[i];
122       byteArray[i] = byteArray[byteArray.length - i - 1];
123       byteArray[byteArray.length - i - 1] = temp;
124     }
125   }
126 
127 }
