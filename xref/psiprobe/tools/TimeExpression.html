11  package psiprobe.tools;
12  
13  /**
14   * The Class TimeExpression.
15   */
16  public final class TimeExpression {
17  
18    /**
19     * Prevent Instantiation.
20     */
21    private TimeExpression() {
22      // Prevent Instantiation
23    }
24  
25    /**
26     * Data points.
27     *
28     * @param periodExpression the period expression
29     * @param spanExpression the span expression
30     *
31     * @return the long
32     */
33    public static long dataPoints(String periodExpression, String spanExpression) {
34      return dataPoints(inSeconds(periodExpression), inSeconds(spanExpression));
35    }
36  
37    /**
38     * Data points.
39     *
40     * @param period the period
41     * @param span the span
42     *
43     * @return the long
44     */
45    public static long dataPoints(long period, long span) {
46      if (period <= 0) {
47        return 0;
48      }
49      return span / period;
50    }
51  
52    /**
53     * Cron expression.
54     *
55     * @param periodExpression the period expression
56     * @param phaseExpression the phase expression
57     *
58     * @return the string
59     */
60    public static String cronExpression(String periodExpression, String phaseExpression) {
61      return cronExpression(inSeconds(periodExpression), inSeconds(phaseExpression));
62    }
63  
64    /**
65     * Cron expression.
66     *
67     * @param period the period
68     * @param phase the phase
69     *
70     * @return the string
71     */
72    public static String cronExpression(long period, long phase) {
73      while (phase >= period) {
74        phase = phase - period;
75      }
76      long secondsPeriod = 0;
77      long minutesPeriod = 0;
78      long hoursPeriod = 0;
79      if (period < 60) {
80        secondsPeriod = period;
81      } else if (period < 60 * 60) {
82        minutesPeriod = period / 60;
83      } else if (period < 60 * 60 * 24) {
84        hoursPeriod = period / (60 * 60);
85      } else {
86        throw new IllegalArgumentException("Period is too large: " + period);
87      }
88      long secondsPhase = 0;
89      long minutesPhase = 0;
90      long hoursPhase = 0;
91      if (phase < 60) {
92        secondsPhase = phase;
93      } else if (phase < 60 * 60) {
94        minutesPhase = phase / 60;
95      } else if (phase < 60 * 60 * 24) {
96        hoursPhase = phase / (60 * 60);
97      } else {
98        throw new IllegalArgumentException("Phase is too large: " + phase);
99      }
100     String secondsCron = cronSubexpression(secondsPeriod, secondsPhase);
101     String minutesCron = "*";
102     String hoursCron = "*";
103     String daysCron = "*";
104     String monthsCron = "*";
105     String dowCron = "?";
106     if (secondsPeriod == 0) {
107       minutesCron = cronSubexpression(minutesPeriod, minutesPhase);
108       if (minutesPeriod == 0) {
109         hoursCron = cronSubexpression(hoursPeriod, hoursPhase);
110       }
111     }
112     return secondsCron + " " + minutesCron + " " + hoursCron + " " + daysCron + " " + monthsCron
113         + " " + dowCron;
114   }
115 
116   /**
117    * Cron subexpression.
118    *
119    * @param period the period
120    * @param phase the phase
121    *
122    * @return the string
123    */
124   private static String cronSubexpression(long period, long phase) {
125     if (period == 0) {
126       return Long.toString(phase);
127     }
128     if (period == 1 && phase == 0) {
129       return "*";
130     }
131     return phase + "/" + period;
132   }
133 
134   /**
135    * In seconds.
136    *
137    * @param expression the expression
138    *
139    * @return the long
140    */
141   public static long inSeconds(String expression) {
142     if (expression == null || expression.isEmpty()) {
143       return 0;
144     }
145     if (expression.matches("[0-9]+[smhd]")) {
146       long multiplier = multiplier(expression.charAt(expression.length() - 1));
147       if (multiplier == 0) {
148         throw new IllegalArgumentException("Invalid unit in expression: " + expression);
149       }
150       long value = Integer.parseInt(expression.substring(0, expression.length() - 1));
151       if (value < 0) {
152         throw new IllegalArgumentException("Invalid value in expression: " + expression);
153       }
154       return value * multiplier;
155     }
156     throw new IllegalArgumentException("Invalid expression format: " + expression);
157   }
158 
159   /**
160    * Multiplier.
161    *
162    * @param unit the unit
163    *
164    * @return the int
165    */
166   private static int multiplier(char unit) {
167     switch (unit) {
168       case 's':
169         return 1;
170       case 'm':
171         return 60;
172       case 'h':
173         return 60 * 60;
174       default:
175         throw new IllegalArgumentException("Invalid unit: " + unit);
176     }
177   }
