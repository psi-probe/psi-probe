27    /** The Constant SIZE_VOID. */
28    public static final long SIZE_VOID = 0;
29  
30    /** The Constant SIZE_BOOLEAN. */
31    public static final long SIZE_BOOLEAN = 1;
32  
33    /** The Constant SIZE_BYTE. */
34    public static final long SIZE_BYTE = 1;
35  
36    /** The Constant SIZE_CHAR. */
37    public static final long SIZE_CHAR = 2;
38  
39    /** The Constant SIZE_SHORT. */
40    public static final long SIZE_SHORT = 2;
41  
42    /** The Constant SIZE_INT. */
43    public static final long SIZE_INT = 4;
44  
45    /** The Constant SIZE_LONG. */
46    public static final long SIZE_LONG = 8;
47  
48    /** The Constant SIZE_FLOAT. */
49    public static final long SIZE_FLOAT = 4;
50  
51    /** The Constant SIZE_DOUBLE. */
52    public static final long SIZE_DOUBLE = 8;
53  
54    /** The Constant SIZE_OBJECT. */
55    public static final long SIZE_OBJECT = 8;
56  
57    /** The Constant SIZE_REFERENCE. */
58    public static final long SIZE_REFERENCE;

60    /** The Constant ACCESSOR. */
61    private static final Accessor ACCESSOR = new SimpleAccessor();
62  
63    /** The Constant IGNORE_NIO. */
64    private static final boolean IGNORE_NIO;
65  
66    static {
67      String ignoreNioProp = System.getProperty("psiprobe.intruments.ignoreNIO");
68      String os64bitProp = System.getProperty("sun.arch.data.model");
69      IGNORE_NIO = ignoreNioProp == null || "true".equalsIgnoreCase(ignoreNioProp);
70      SIZE_REFERENCE = "64".equals(os64bitProp) ? 8 : 4;
71    }

73    /** The processed objects. */
74    private Set<Object> processedObjects = new HashSet<>(2048);

78

80    private final List<Object> nextQueue = new LinkedList<>();

124   /**
125    * Internal size of.
126    *
127    * @param root the root
128    *
129    * @return the long
130    */
131   private long internalSizeOf(Object root) {
132     long size = 0;
133     thisQueue.add(root);
134     while (!thisQueue.isEmpty()) {
135       Iterator<Object> it = thisQueue.iterator();
136       while (it.hasNext()) {
137         Object obj = it.next();
138         if (isInitialized() && obj != null
139             && (classLoader == null || classLoader == obj.getClass().getClassLoader())
140             && (!IGNORE_NIO || !obj.getClass().getName().startsWith("java.nio."))) {
141           ObjectWrapper ow = new ObjectWrapper(obj);
142           if (!processedObjects.contains(ow)) {
143             if (obj.getClass().isArray()) {
144               size += sizeOfArray(obj);
145             } else if (obj.getClass().isPrimitive()) {
146               size += sizeOfPrimitive(obj.getClass());
147             } else {
148               processedObjects.add(ow);
149               size += sizeOfObject(obj);
150             }
151           }
152         }
153         it.remove();
154       }
155       // avoids ConcurrentModificationException
156       if (!nextQueue.isEmpty()) {
157         thisQueue.addAll(nextQueue);
158         nextQueue.clear();
159       }
160     }
161     return size;
162   }
163 
164   /**
165    * Size of object.
166    *
167    * @param obj the obj
168    *
169    * @return the long
170    */
171   private long sizeOfObject(Object obj) {
172     long size = SIZE_OBJECT;
173     Class<? extends Object> clazz = obj.getClass();
174     while (clazz != null) {
175       Field[] fields = clazz.getDeclaredFields();
176       for (Field field : fields) {
177         if (!Modifier.isStatic(field.getModifiers())) {
178           if (field.getType().isPrimitive()) {
179             size += sizeOfPrimitive(field.getType());
180           } else {
181             Object val = ACCESSOR.get(obj, field);
182             if (field.getType().isArray()) {
183               size += sizeOfArray(val);
184             } else {
185               size += SIZE_REFERENCE;
186               nextQueue.add(val);
187             }
188           }
189         }
190       }
191       clazz = clazz.getSuperclass();
192     }
193     return size;
194   }
195 
196   /**
197    * Size of array.
198    *
199    * @param obj the obj
200    *
201    * @return the long
202    */
203   private long sizeOfArray(Object obj) {
204     if (obj != null) {
205       Class<?> ct = obj.getClass().getComponentType();
206       if (ct.isPrimitive()) {
207         return Array.getLength(obj) * sizeOfPrimitive(ct);
208       }
209       for (int i = 0; i < Array.getLength(obj); i++) {
210         nextQueue.add(Array.get(obj, i));
211       }
212     }
213     return 0;
214   }

216   /**
217    * Size of primitive.
218    *
219    * @param <T> the generic type
220    * @param type the type
221    *
222    * @return the long
223    */
224   private static <T> long sizeOfPrimitive(Class<T> type) {
225     if (type == Boolean.TYPE) {
226       return SIZE_BOOLEAN;
227     }
228     if (type == Byte.TYPE) {
229       return SIZE_BYTE;
230     }
231     if (type == Character.TYPE) {
232       return SIZE_CHAR;
233     }
234     if (type == Short.TYPE) {
235       return SIZE_SHORT;
236     }
237     if (type == Integer.TYPE) {
238       return SIZE_INT;
239     }
240     if (type == Long.TYPE) {
241       return SIZE_LONG;
242     }
243     if (type == Float.TYPE) {
244       return SIZE_FLOAT;
245     }
246     if (type == Double.TYPE) {
247       return SIZE_DOUBLE;
248     }
249     if (type == Void.TYPE) {
250       return SIZE_VOID;
251     }
252     return SIZE_REFERENCE;
253   }

264   /**
265    * Gets the field.
266    *
267    * @param obj the obj
268    * @param name the name
269    *
270    * @return the field
271    */
272   public static Object getField(Object obj, String name) {
273     if (isInitialized()) {
274       Field field = findField(obj.getClass(), name);
275       if (field != null) {
276         return ACCESSOR.get(obj, field);
277       }
278     }
279     return null;
280   }
281 
282   /**
283    * Find field.
284    *
285    * @param <T> the generic type
286    * @param clazz the clazz
287    * @param name the name
288    *
289    * @return the field
290    */
291   public static <T> Field findField(Class<T> clazz, String name) {
292     Field[] fields = clazz.getDeclaredFields();
293     for (Field field : fields) {
294       if (name.equals(field.getName())) {
295         return field;
296       }
297     }
298     Class<?> superClass = clazz.getSuperclass();
299     if (superClass != null) {
300       return findField(superClass, name);
301     }
302     return null;
303   }
