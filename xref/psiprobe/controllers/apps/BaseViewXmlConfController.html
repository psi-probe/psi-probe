11  package psiprobe.controllers.apps;
12  
13  import java.io.File;
14  import java.io.InputStream;
15  import java.nio.charset.Charset;
16  import java.nio.file.Files;
17  
18  import javax.servlet.ServletContext;
19  import javax.servlet.http.HttpServletRequest;
20  import javax.servlet.http.HttpServletResponse;
21  
22  import org.apache.catalina.Context;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  import org.springframework.web.servlet.ModelAndView;
26  
27  import psiprobe.Utils;
28  import psiprobe.controllers.AbstractContextHandlerController;
29  
30  /**
31   * Displays a deployment descriptor (web.xml) or a context descriptor (context.xml) of a web
32   * application
33   */
34  public class BaseViewXmlConfController extends AbstractContextHandlerController {
35  
36    /** The Constant logger. */
37    private static final Logger logger = LoggerFactory.getLogger(BaseViewXmlConfController.class);
38  
39    /** The Constant TARGET_WEB_XML. */
40    private static final String TARGET_WEB_XML = "web.xml";
41  
42    /** The Constant TARGET_CONTEXT_XML. */
43    private static final String TARGET_CONTEXT_XML = "context.xml";
44  
45    /** Type of a file to be displayed. */
46    private String displayTarget;
47  
48    /** Url that will be used in the view to download the file. */
49    private String downloadUrl;
50  
51    /**
52     * Gets the display target.
53     *
54     * @return the display target
55     */
56    public String getDisplayTarget() {
57      return displayTarget;
58    }
59  
60    /**
61     * Sets the display target.
62     *
63     * @param displayTarget the new display target
64     */
65    public void setDisplayTarget(String displayTarget) {
66      this.displayTarget = displayTarget;
67    }
68  
69    /**
70     * Gets the download url.
71     *
72     * @return the download url
73     */
74    public String getDownloadUrl() {
75      return downloadUrl;
76    }
77  
78    /**
79     * Sets the download url.
80     *
81     * @param downloadUrl the new download url
82     */
83    public void setDownloadUrl(String downloadUrl) {
84      this.downloadUrl = downloadUrl;
85    }
86  
87    @Override
88    protected ModelAndView handleContext(String contextName, Context context,
89        HttpServletRequest request, HttpServletResponse response) throws Exception {
90  
91      if (displayTarget == null) {
92        throw new RuntimeException("Display target is not set for " + getClass().getName());
93      }
94  
95      String xmlPath;
96      File xmlFile = null;
97      ModelAndView mv = new ModelAndView(getViewName());
98  
99      if (TARGET_WEB_XML.equals(displayTarget)) {
100       ServletContext sctx = context.getServletContext();
101       xmlPath = sctx.getRealPath("/WEB-INF/web.xml");
102       xmlFile = new File(xmlPath);
103       mv.addObject("fileDesc",
104           getMessageSourceAccessor().getMessage("probe.src.app.viewxmlconf.webxml.desc"));
105     } else if (TARGET_CONTEXT_XML.equals(displayTarget)) {
106       xmlFile = getContainerWrapper().getTomcatContainer().getConfigFile(context);
107       if (xmlFile != null) {
108         xmlPath = xmlFile.getPath();
109       } else {
110         xmlPath = null;
111       }
112       mv.addObject("fileDesc",
113           getMessageSourceAccessor().getMessage("probe.src.app.viewxmlconf.contextxml.desc"));
114     } else {
115       throw new RuntimeException("Unknown display target " + getDisplayTarget());
116     }
117 
118     mv.addObject("displayTarget", displayTarget);
119     mv.addObject("downloadUrl", downloadUrl);
120 
121     if (xmlFile != null) {
122       mv.addObject("fileName", xmlFile.getName());
123       if (xmlFile.exists()) {
124         try (InputStream fis = Files.newInputStream(xmlFile.toPath())) {
125           String encoding = Charset.defaultCharset().displayName();
126           mv.addObject("content", Utils.highlightStream(TARGET_WEB_XML, fis, "xml",
127               encoding == null ? "ISO-8859-1" : encoding));
128         }
129       } else {
130         logger.debug("File {} of {} application does not exists.", xmlPath, contextName);
131       }
132     } else {
133       logger.debug("Cannot determine path to {} file of {} application.", getDisplayTarget(),
134           contextName);
135     }
136 
137     return mv;
138   }
139 
140 }
