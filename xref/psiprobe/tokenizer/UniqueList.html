11  package psiprobe.tokenizer;
12  
13  import java.util.ArrayList;
14  import java.util.Collection;
15  import java.util.Collections;
16  import java.util.Comparator;
17  
18  /**
19   * <code>UniqueList</code> is a successor of <code>java.util.Vector</code> to provide a collection
20   * that contains no duplicate elements, more formally such that e1.compareTo(e2) == 0.
21   * 
22   * <p>
23   * The collection is kept ordered whenever elements added or removed and besides uniqueness it is to
24   * provide fast element search based again on e1.compareTo(e2) values.
25   * </p>
26   *
27   * @param <T> the generic type
28   */
29  public class UniqueList<T extends Comparable<? super T>> extends ArrayList<T> {
30  
31    /** The Constant serialVersionUID. */
32    private static final long serialVersionUID = 1L;
33  
34    @Override
35    public synchronized boolean add(T obj) {
36      return add(obj, null);
37    }
38  
39    /**
40     * Adds the.
41     *
42     * @param obj the obj
43     * @param comp the comp
44     * @return true, if successful
45     */
46    protected synchronized boolean add(T obj, Comparator<? super T> comp) {
47      if (size() == 0) {
48        return super.add(obj);
49      }
50      int index;
51      index = comp == null ? Collections.binarySearch(this, obj)
52          : Collections.binarySearch(this, obj, comp);
53      if (index < 0) {
54        int insertionPoint = -index - 1;
55        if (insertionPoint >= size()) {
56          super.add(obj);
57        } else {
58          super.add(insertionPoint, obj);
59        }
60      }
61      return index < 0;
62    }
63  
64    @Override
65    public synchronized void add(int index, T obj) {
66      add(obj);
67    }
68  
69    @Override
70    public synchronized boolean addAll(Collection<? extends T> comp) {
71      boolean ok = this != comp;
72      if (ok) {
73        for (T compItem : comp) {
74          ok &= this.add(compItem);
75        }
76      }
77      return ok;
78    }
79  
80  }
