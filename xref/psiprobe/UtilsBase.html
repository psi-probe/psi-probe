11  package psiprobe;
12  
13  import java.util.Scanner;
14  
15  /**
16   * Misc. static helper methods.
17   */
18  public class UtilsBase {
19  
20    /**
21     * Prevent Instantiation.
22     */
23    private UtilsBase() {
24      // Prevent Instantiation
25    }
26  
27    /**
28     * Calc pool usage score.
29     *
30     * @param max the max
31     * @param value the value
32     * @return the int
33     */
34    public static int calcPoolUsageScore(int max, int value) {
35      return max > 0 ? Math.max(0, value) * 100 / max : 0;
36    }
37  
38    /**
39     * To int.
40     *
41     * @param num the num
42     * @param defaultValue the default value
43     * @return the int
44     */
45    public static int toInt(String num, int defaultValue) {
46      if (num != null && !num.contains(" ")) {
47        try (Scanner scanner = new Scanner(num)) {
48          if (scanner.hasNextInt()) {
49            return Integer.parseInt(num);
50          }
51        }
52      }
53      return defaultValue;
54    }
55  
56  }
