94    @Override
95    public void setWrapper(Wrapper wrapper) {
96      Valve valve = createValve();
97      if (wrapper != null) {
98        host = (Host) wrapper.getParent().getParent();
99        Engine engine = (Engine) host.getParent();
100       Service service = engine.getService();
101       connectors = service.findConnectors();
102       try {
103         deployerOName =
104             new ObjectName(host.getParent().getName() + ":type=Deployer,host=" + host.getName());
105       } catch (MalformedObjectNameException e) {
106         logger.trace("", e);
107       }
108       host.getPipeline().addValve(valve);
109       mbeanServer = ManagementFactory.getPlatformMBeanServer();
110     } else if (host != null) {
111       host.getPipeline().removeValve(valve);
112     }
113   }

124   @Override
125   public String getConfigBase() {
126     File configBase = new File(System.getProperty("catalina.base"), "conf");
127     Container baseHost = null;
128     Container thisContainer = host;
129     while (thisContainer != null) {
130       if (thisContainer instanceof Host) {
131         baseHost = thisContainer;
132       }
133       thisContainer = thisContainer.getParent();
134     }
135     if (baseHost != null) {
136       configBase = new File(configBase, baseHost.getName());
137     }
138     return configBase.getAbsolutePath();
139   }

190   @Override
191   public void remove(String name) throws Exception {
192     name = formatContextName(name);
193     Context ctx = findContext(name);
194 
195     if (ctx != null) {
196 
197       try {
198         stop(name);
199       } catch (Exception e) {
200         logger.info("Stopping '{}' threw this exception:", name, e);
201       }
202 
203       File appDir;
204       File docBase = new File(ctx.getDocBase());
205 
206       if (!docBase.isAbsolute()) {
207         appDir = new File(getAppBase(), ctx.getDocBase());
208       } else {
209         appDir = docBase;
210       }
211 
212       logger.debug("Deleting '{}'", appDir.getAbsolutePath());
213       Utils.delete(appDir);
214 
215       String warFilename = formatContextFilename(name);
216       File warFile = new File(getAppBase(), warFilename + ".war");
217       logger.debug("Deleting '{}'", warFile.getAbsolutePath());
218       Utils.delete(warFile);
219 
220       File configFile = getConfigFile(ctx);
221       if (configFile != null) {
222         logger.debug("Deleting '{}'", configFile.getAbsolutePath());
223         Utils.delete(configFile);
224       }
225 
226       removeInternal(name);
227     }
228   }

255   @Override
256   public Context findContext(String name) {
257     String safeName = formatContextName(name);
258     if (safeName == null) {
259       return null;
260     }
261     Context result = findContextInternal(safeName);
262     if (result == null && "".equals(safeName)) {
263       result = findContextInternal("/");
264     }
265     return result;
266   }

268   @Override
269   public String formatContextName(String name) {
270     if (name == null) {
271       return null;
272     }
273     String result = name.trim();
274     if (!result.startsWith("/")) {
275       result = "/" + result;
276     }
277     if ("/".equals(result) || "/ROOT".equals(result)) {
278       result = "";
279     }
280     // For ROOT Parallel Deployment, remove "/ROOT"
281     if (result.startsWith("/ROOT##")) {
282       result = result.substring(5);
283     }
284     // For ROOT Parallel Usage, remove "/"
285     if (result.startsWith("/##")) {
286       result = result.substring(1);
287     }
288     return result;
289   }

291   @Override
292   public String formatContextFilename(String contextName) {
293     if (contextName == null) {
294       return null;
295     }
296     if ("".equals(contextName)) {
297       return "ROOT";
298     }
299     if (contextName.startsWith("/")) {
300       return contextName.substring(1);
301     }
302     return contextName;
303   }

318   @Override
319   public String getServletFileNameForJsp(Context context, String jspName) {
320     String servletName = null;
321 
322     ServletConfig servletConfig = (ServletConfig) context.findChild("jsp");
323     if (servletConfig != null) {
324       ServletContext sctx = context.getServletContext();
325       Options opt = new EmbeddedServletOptions(servletConfig, sctx);
326       JspRuntimeContext jrctx = new JspRuntimeContext(sctx, opt);
327       JspCompilationContext jcctx = createJspCompilationContext(jspName, opt, sctx, jrctx, null);
328       servletName = jcctx.getServletJavaFileName();
329     } else {
330       logger.error(NO_JSP_SERVLET, context.getName());
331     }
332     return servletName;
333   }

335   @Override
336   public void recompileJsps(Context context, Summary summary, List<String> names) {
337     ServletConfig servletConfig = (ServletConfig) context.findChild("jsp");
338     if (servletConfig != null) {
339       if (summary != null) {
340         synchronized (servletConfig) {
341           ServletContext sctx = context.getServletContext();
342           Options opt = new EmbeddedServletOptions(servletConfig, sctx);
343 
344           JspRuntimeContext jrctx = new JspRuntimeContext(sctx, opt);
345           /*
346            * we need to pass context classloader here, so the jsps can reference /WEB-INF/classes
347            * and /WEB-INF/lib. JspCompilationContext would only take URLClassLoader, so we fake it
348            */
349           try (URLClassLoader classLoader =
350               new URLClassLoader(new URL[0], context.getLoader().getClassLoader())) {
351             for (String name : names) {
352               long time = System.currentTimeMillis();
353               JspCompilationContext jcctx =
354                   createJspCompilationContext(name, opt, sctx, jrctx, classLoader);
355               ClassLoader prevCl = ClassUtils.overrideThreadContextClassLoader(classLoader);
356               try {
357                 Item item = summary.getItems().get(name);
358                 if (item != null) {
359                   try {
360                     org.apache.jasper.compiler.Compiler compiler = jcctx.createCompiler();
361                     compiler.compile();
362                     item.setState(Item.STATE_READY);
363                     item.setException(null);
364                     logger.info("Compiled '{}': OK", name);
365                   } catch (Exception e) {
366                     item.setState(Item.STATE_FAILED);
367                     item.setException(e);
368                     logger.error("Compiled '{}': FAILED", name, e);
369                   }
370                   item.setCompileTime(System.currentTimeMillis() - time);
371                 } else {
372                   logger.error("{} is not on the summary list, ignored", name);
373                 }
374               } finally {
375                 ClassUtils.overrideThreadContextClassLoader(prevCl);
376               }
377             }
378           } catch (IOException e) {
379             this.logger.error("", e);
380           } finally {
381             jrctx.destroy();
382           }
383         }
384       } else {
385         logger.error("summary is null for '{}', request ignored", context.getName());
386       }
387     } else {
388       logger.error(NO_JSP_SERVLET, context.getName());
389     }
390   }

392   @Override
393   public void listContextJsps(Context context, Summary summary, boolean compile) {
394     ServletConfig servletConfig = (ServletConfig) context.findChild("jsp");
395     if (servletConfig != null) {
396       synchronized (servletConfig) {
397         ServletContext sctx = context.getServletContext();
398         Options opt = new EmbeddedServletOptions(servletConfig, sctx);
399 
400         JspRuntimeContext jrctx = new JspRuntimeContext(sctx, opt);
401         try {
402           if (summary.getItems() == null) {
403             summary.setItems(new HashMap<>());
404           }
405 
406           /*
407            * mark all items as missing
408            */
409           for (Item item : summary.getItems().values()) {
410             item.setMissing(true);
411           }
412 
413           /*
414            * we need to pass context classloader here, so the jsps can reference /WEB-INF/classes
415            * and /WEB-INF/lib. JspCompilationContext would only take URLClassLoader, so we fake it
416            */
417           try (URLClassLoader urlcl =
418               new URLClassLoader(new URL[0], context.getLoader().getClassLoader())) {
419 
420             compileItem("/", opt, context, jrctx, summary, urlcl, 0, compile);
421           } catch (IOException e) {
422             this.logger.error("", e);
423           }
424         } finally {
425           jrctx.destroy();
426         }
427       }
428 
429       //
430       // delete "missing" items by keeping "not missing" ones
431       //
432       Map<String, Item> hashMap = new HashMap<>();
433       for (String key : summary.getItems().keySet()) {
434         Item item = summary.getItems().get(key);
435         if (!item.isMissing()) {
436           hashMap.put(key, item);
437         }
438       }
439 
440       summary.setItems(hashMap);
441     } else {
442       logger.error(NO_JSP_SERVLET, context.getName());
443     }
444   }

451   @Override
452   public File getConfigFile(Context context) {
453     URL configUrl = context.getConfigFile();
454     if (configUrl != null) {
455       try {
456         URI configUri = configUrl.toURI();
457         if ("file".equals(configUri.getScheme())) {
458           return new File(configUri.getPath());
459         }
460       } catch (URISyntaxException ex) {
461         logger.error("Could not convert URL to URI: '{}'", configUrl, ex);
462       }
463     }
464     return null;
465   }

487   /**
488    * Change context binding.
489    *
490    * @param context the context
491    * @param bind the bind
492    * @throws NamingException the naming exception
493    */
494   private void changeContextBinding(Context context, boolean bind) throws NamingException {
495     Object token = getNamingToken(context);
496     ClassLoader loader = Thread.currentThread().getContextClassLoader();
497     if (bind) {
498       ContextBindings.bindClassLoader(context, token, loader);
499     } else {
500       ContextBindings.unbindClassLoader(context, token, loader);
501     }
502   }

504   /**
505    * Lists and optionally compiles a directory recursively.
506    *
507    * @param jspName name of JSP file or directory to be listed and compiled.
508    * @param opt the JSP compiler options
509    * @param ctx the context
510    * @param jrctx the runtime context used to create the compilation context
511    * @param summary the summary in which the output is stored
512    * @param classLoader the classloader used by the compiler
513    * @param level the depth in the tree at which the item was encountered
514    * @param compile whether or not to compile the item or just to check whether it's out of date
515    */
516   protected void compileItem(String jspName, Options opt, Context ctx, JspRuntimeContext jrctx,
517       Summary summary, URLClassLoader classLoader, int level, boolean compile) {
518     ServletContext sctx = ctx.getServletContext();
519     Set<String> paths = sctx.getResourcePaths(jspName);
520 
521     if (paths != null) {
522       for (String name : paths) {
523         boolean isJsp = false;
524 
525         try {
526           isJsp =
527               name.endsWith(".jsp") || name.endsWith(".jspx") || opt.getJspConfig().isJspPage(name);
528         } catch (Exception e) {
529           // XXX Tomcat 7.0.x throws JasperException otherwise this could be removed.
530           logger.info("isJspPage() thrown an error for '{}'", name, e);
531         }
532 
533         if (isJsp) {
534           JspCompilationContext jcctx =
535               createJspCompilationContext(name, opt, sctx, jrctx, classLoader);
536           ClassLoader prevCl = ClassUtils.overrideThreadContextClassLoader(classLoader);
537           try {
538             Item item = summary.getItems().get(name);
539 
540             if (item == null) {
541               item = new Item();
542               item.setName(name);
543             }
544 
545             item.setLevel(level);
546             item.setCompileTime(-1);
547 
548             Long[] objects = this.getResourceAttributes(name, ctx);
549             item.setSize(objects[0]);
550             item.setLastModified(objects[1]);
551 
552             long time = System.currentTimeMillis();
553             try {
554               org.apache.jasper.compiler.Compiler compiler = jcctx.createCompiler();
555               if (compile) {
556                 compiler.compile();
557                 item.setState(Item.STATE_READY);
558                 item.setException(null);
559               } else if (!compiler.isOutDated()) {
560                 item.setState(Item.STATE_READY);
561                 item.setException(null);
562               } else if (item.getState() != Item.STATE_FAILED) {
563                 item.setState(Item.STATE_OOD);
564                 item.setException(null);
565               }
566               logger.info("Compiled '{}': OK", name);
567             } catch (Exception e) {
568               item.setState(Item.STATE_FAILED);
569               item.setException(e);
570               logger.info("Compiled '{}': FAILED", name, e);
571             }
572             if (compile) {
573               item.setCompileTime(System.currentTimeMillis() - time);
574             }
575             item.setMissing(false);
576             summary.getItems().put(name, item);
577           } finally {
578             ClassUtils.overrideThreadContextClassLoader(prevCl);
579           }
580         } else {
581           compileItem(name, opt, ctx, jrctx, summary, classLoader, level + 1, compile);
582         }
583       }
584     } else {
585       logger.debug("getResourcePaths() is null for '{}'. Empty dir? Or Tomcat bug?", jspName);
586     }
587   }

599   /**
600    * Check changes.
601    *
602    * @param name the name
603    * @throws Exception the exception
604    */
605   protected void checkChanges(String name) throws Exception {
606     Boolean result = (Boolean) mbeanServer.invoke(deployerOName, "isServiced", new String[] {name},
607         new String[] {String.class.getName()});
608     if (!result.booleanValue()) {
609       mbeanServer.invoke(deployerOName, "addServiced", new String[] {name},
610           new String[] {String.class.getName()});
611       try {
612         mbeanServer.invoke(deployerOName, "check", new String[] {name},
613             new String[] {String.class.getName()});
614       } finally {
615         mbeanServer.invoke(deployerOName, "removeServiced", new String[] {name},
616             new String[] {String.class.getName()});
617       }
618     }
619   }

649   /**
650    * Adds the filter mapping.
651    *
652    * @param filterName the filter name
653    * @param dispatcherMap the dispatcher map
654    * @param filterClass the filter class
655    * @param types the types as urls or servlet name
656    * @param results the results
657    * @param filterMapType the filter map type
658    */
659   protected void addFilterMapping(String filterName, String dispatcherMap, String filterClass,
660       String[] types, List<FilterMapping> results, FilterMapType filterMapType) {
661     for (String type : types) {
662       FilterMapping fm = new FilterMapping();
663       if (filterMapType == FilterMapType.URL) {
664         fm.setUrl(type);
665       } else {
666         fm.setServletName(type);
667       }
668       fm.setFilterName(filterName);
669       fm.setDispatcherMap(dispatcherMap);
670       fm.setFilterClass(filterClass);
671       results.add(fm);
672     }
673   }
