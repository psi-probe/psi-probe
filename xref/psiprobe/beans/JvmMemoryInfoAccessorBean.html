11  package psiprobe.beans;
12  
13  import java.lang.management.ManagementFactory;
14  import java.util.LinkedList;
15  import java.util.List;
16  import java.util.Set;
17  
18  import javax.management.MBeanServer;
19  import javax.management.ObjectInstance;
20  import javax.management.ObjectName;
21  import javax.management.openmbean.CompositeDataSupport;
22  
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import psiprobe.model.jmx.MemoryPool;
27  import psiprobe.tools.JmxTools;
28  
29  /**
30   * The Class JvmMemoryInfoAccessorBean.
31   */
32  public class JvmMemoryInfoAccessorBean {
33  
34    /** The Constant logger. */
35    private static final Logger logger = LoggerFactory.getLogger(JvmMemoryInfoAccessorBean.class);
36  
37    /**
38     * Gets the pools.
39     *
40     * @return the pools
41     *
42     * @throws Exception the exception
43     */
44    public List<MemoryPool> getPools() throws Exception {
45  
46      List<MemoryPool> memoryPools = new LinkedList<>();
47      MBeanServer mbeanServer = ManagementFactory.getPlatformMBeanServer();
48      Set<ObjectInstance> memoryOPools =
49          mbeanServer.queryMBeans(new ObjectName("java.lang:type=MemoryPool,*"), null);
50  
51      // totals
52      long totalInit = 0;
53      long totalMax = 0;
54      long totalUsed = 0;
55      long totalCommitted = 0;
56  
57      for (ObjectInstance oi : memoryOPools) {
58        ObjectName objName = oi.getObjectName();
59        MemoryPool memoryPool = new MemoryPool();
60        memoryPool.setName(JmxTools.getStringAttr(mbeanServer, objName, "Name"));
61        memoryPool.setType(JmxTools.getStringAttr(mbeanServer, objName, "Type"));
62  
63        CompositeDataSupport cd = (CompositeDataSupport) mbeanServer.getAttribute(objName, "Usage");
64        /*
65         * It seems that "Usage" attribute of one of the pools may turn into null intermittently. We
66         * better have a dip in the graph then an NPE though.
67         */
68        if (cd != null) {
69          memoryPool.setMax(JmxTools.getLongAttr(cd, "max"));
70          memoryPool.setUsed(JmxTools.getLongAttr(cd, "used"));
71          memoryPool.setInit(JmxTools.getLongAttr(cd, "init"));
72          memoryPool.setCommitted(JmxTools.getLongAttr(cd, "committed"));
73        } else {
74          logger.error("Oops, JVM problem? {} 'Usage' attribute is NULL!", objName);
75        }
76  
77        totalInit += memoryPool.getInit();
78        totalMax += memoryPool.getMax();
79        totalUsed += memoryPool.getUsed();
80        totalCommitted += memoryPool.getCommitted();
81  
82        memoryPools.add(memoryPool);
83      }
84  
85      if (!memoryPools.isEmpty()) {
86        MemoryPool pool = new MemoryPool();
87        pool.setName("Total");
88        pool.setType("TOTAL");
89        pool.setInit(totalInit);
90        pool.setUsed(totalUsed);
91        pool.setMax(totalMax);
92        pool.setCommitted(totalCommitted);
93        memoryPools.add(pool);
94      }
95  
96      return memoryPools;
97  
98    }
99  }
