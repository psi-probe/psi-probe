34  /**
35   * An Adapter to convert information retrieved from JBoss JMX beans into internal resource model.
36   */
37  public class JBossResourceResolverBean implements ResourceResolver {
38  
39    /** The Constant logger. */
40    private static final Logger logger = LoggerFactory.getLogger(JBossResourceResolverBean.class);
41  
42    @Override
43    public MBeanServer getMBeanServer() {
44      for (MBeanServer server : MBeanServerFactory.findMBeanServer(null)) {
45        if ("jboss".equals(server.getDefaultDomain())
46            || "DefaultDomain".equals(server.getDefaultDomain())) {
47          return server;
48        }
49      }
50      return null;
51    }
52  
53    @Override
54    public boolean supportsPrivateResources() {
55      return false;
56    }
57  
58    @Override
59    public boolean supportsGlobalResources() {
60      return true;
61    }
62  
63    @Override
64    public boolean supportsDataSourceLookup() {
65      return false;
66    }

68    @Override
69    public List<ApplicationResource> getApplicationResources() throws NamingException {
70  
71      List<ApplicationResource> resources = new ArrayList<>();
72  
73      MBeanServer server = getMBeanServer();
74      if (server != null) {
75        try {
76          Set<ObjectName> dsNames =
77              server.queryNames(new ObjectName("jboss.jca:service=ManagedConnectionPool,*"), null);
78          for (ObjectName managedConnectionPoolOName : dsNames) {
79            ApplicationResource resource = new ApplicationResource();
80            resource.setName(managedConnectionPoolOName.getKeyProperty("name"));
81            resource.setType("jboss");
82            String criteria = (String) server.getAttribute(managedConnectionPoolOName, "Criteria");
83            if ("ByApplication".equals(criteria)) {
84              resource.setAuth("Application");
85            } else if ("ByContainerAndApplication".equals(criteria)) {
86              resource.setAuth("Both");
87            } else {
88              resource.setAuth("Container");
89            }
90            DataSourceInfo dsInfo = new DataSourceInfo();
91            dsInfo.setMaxConnections(
92                (Integer) server.getAttribute(managedConnectionPoolOName, "MaxSize"));
93            dsInfo.setEstablishedConnections(
94                (Integer) server.getAttribute(managedConnectionPoolOName, "ConnectionCount"));
95            dsInfo.setBusyConnections(
96                ((Long) server.getAttribute(managedConnectionPoolOName, "InUseConnectionCount"))
97                    .intValue());
98            ObjectName connectionFactoryOName = new ObjectName(
99                "jboss.jca:service=ManagedConnectionFactory,name=" + resource.getName());
100           Element elm = (Element) server.getAttribute(connectionFactoryOName,
101               "ManagedConnectionFactoryProperties");
102 
103           if (elm != null) {
104             NodeList nl = elm.getChildNodes();
105             for (int i = 0; i < nl.getLength(); i++) {
106               Node node = nl.item(i);
107               Node na = node.getAttributes().getNamedItem("name");
108               if (na != null) {
109                 if ("ConnectionURL".equals(na.getNodeValue())) {
110                   dsInfo.setJdbcUrl(node.getFirstChild().getNodeValue());
111                 }
112 
113                 if ("UserName".equals(na.getNodeValue())) {
114                   dsInfo.setUsername(node.getFirstChild().getNodeValue());
115                 }
116 
117                 // JMS datasource
118                 if ("JmsProviderAdapterJNDI".equals(na.getNodeValue())) {
119                   dsInfo.setJdbcUrl(node.getFirstChild().getNodeValue());
120                   resource.setType("jms");
121                 }
122               }
123             }
124           }
125 
126           dsInfo.setResettable(true);
127 
128           resource.setDataSourceInfo(dsInfo);
129           resources.add(resource);
130         }
131       } catch (Exception e) {
132         logger.error("There was an error querying JBoss JMX server:", e);
133       }
134     }
135     return resources;
136   }

153       ContainerWrapperBean containerWrapper) throws NamingException {
154 
155     throw new UnsupportedOperationException("Not supported yet.");
156   }

158   @Override
159   public boolean resetResource(Context context, String resourceName,
160       ContainerWrapperBean containerWrapper) throws NamingException {
161     try {
162       ObjectName poolOName =
163           new ObjectName("jboss.jca:service=ManagedConnectionPool,name=" + resourceName);
164       MBeanServer server = getMBeanServer();
165       if (server != null) {
166         try {
167           server.invoke(poolOName, "stop", null, null);
168           server.invoke(poolOName, "start", null, null);
169           return true;
170         } catch (Exception e) {
171           logger.error("Could not reset resource '{}'", resourceName, e);
172         }
173       }
174       return false;
175     } catch (MalformedObjectNameException e) {
176       logger.trace("", e);
177       throw new NamingException(
178           "Resource name: \"" + resourceName + "\" makes a malformed ObjectName");
179     }
180   }

182   @Override
183   public DataSource lookupDataSource(Context context, String resourceName,
184       ContainerWrapperBean containerWrapper) throws NamingException {
185     throw new UnsupportedOperationException(
186         "This feature has not been implemented for JBoss server yet.");
187   }
