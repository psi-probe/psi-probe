36    /**
37     * Test range scan.
38     */
39    @Test
40    void testRangeScan() {
41      // As used in appRuntimeInfo.jsp
42      doTestRangeScan(8, 5, false);
43      doTestRangeScan(8, 5, true);
44  
45      /*
46       * As used in memory_pools.jsp, application.jsp, datasourcegroup.jsp, datasources_table.jsp,
47       * resources.jsp and sysinfo.jsp
48       */
49      doTestRangeScan(10, 5, false);
50      doTestRangeScan(10, 5, true);
51    }
52  
53    /**
54     * Do test range scan.
55     *
56     * @param fullBlocks the full blocks
57     * @param partialBlocks the partial blocks
58     * @param invertLoopIndexes the invert loop indexes
59     */
60    private static void doTestRangeScan(int fullBlocks, int partialBlocks,
61        boolean invertLoopIndexes) {
62      int value;
63      int value2;
64      int count = 0;
65      for (int i = 0; i <= 100; i++) {
66        for (int j = 0; j <= 100; j++) {
67          if (invertLoopIndexes) {
68            value = j;
69            value2 = i;
70          } else {
71            value = i;
72            value2 = j;
73          }
74          String[] split = callCalculateSuffix(value, value2, fullBlocks, partialBlocks);
75          for (String suffix : split) {
76            logger.trace(suffix);
77            String[] values = suffix.split("\\+");
78            if (values.length > 1) {
79              value = Integer.parseInt(values[0]);
80              value2 = Integer.parseInt(values[1]);
81              // TODO JWL 12/12/2016 This never occurs so why do we care?
82              if (value > 5 || value2 > 5) {
83                count++;
84                StringBuilder msg = new StringBuilder();
85                msg.append("Found incorrect value ");
86                msg.append(suffix);
87                msg.append(". value = ");
88                msg.append(invertLoopIndexes ? j : i);
89                msg.append(" value2 = ");
90                msg.append(invertLoopIndexes ? i : j);
91                msg.append(" fullBlocks = ");
92                msg.append(fullBlocks);
93                msg.append(" partialBlocks = ");
94                msg.append(partialBlocks);
95                logger.trace(msg.toString());
96              }
97            }
98          }
99        }
100     }
101     Assertions.assertFalse(count > 0, "Incorrect values were founded " + count + " times");
102   }
103 
104   /**
105    * Call calculate suffix.
106    *
107    * @param value the value
108    * @param value2 the value2
109    * @param fullBlocks the full blocks
110    * @param partialBlocks the partial blocks
111    * @return the string[]
112    */
113   private static String[] callCalculateSuffix(int value, int value2, int fullBlocks,
114       int partialBlocks) {
115     String body = "{0} ";
116 
117     VisualScoreTag visualScoreTag = new VisualScoreTag();
118 
119     // Values are based on datasources_table.jsp
120     visualScoreTag.setFullBlocks(fullBlocks);
121     visualScoreTag.setPartialBlocks(partialBlocks);
122     visualScoreTag.setShowEmptyBlocks(true);
123     visualScoreTag.setShowA(true);
124     visualScoreTag.setShowB(true);
125 
126     visualScoreTag.setValue(value);
127     visualScoreTag.setValue2(value2);
128 
129     String buf = visualScoreTag.calculateSuffix(body);
130 
131     return buf.split(" ");
132   }
