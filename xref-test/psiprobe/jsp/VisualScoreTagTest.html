36    /**
37     * Test range scan.
38     */
39    @Test
40    void rangeScanTest() {
41      // As used in appRuntimeInfo.jsp
42      doTestRangeScan(8, 5, false);
43      doTestRangeScan(8, 5, true);
44  
45      /*
46       * As used in memory_pools.jsp, application.jsp, datasourcegroup.jsp, datasources_table.jsp,
47       * resources.jsp and sysinfo.jsp
48       */
49      doTestRangeScan(10, 5, false);
50      doTestRangeScan(10, 5, true);
51    }
52  
53    /**
54     * Do test range scan.
55     *
56     * @param fullBlocks the full blocks
57     * @param partialBlocks the partial blocks
58     * @param invertLoopIndexes the invert loop indexes
59     */
60    private static void doTestRangeScan(int fullBlocks, int partialBlocks,
61        boolean invertLoopIndexes) {
62      int value;
63      int value2;
64      int count = 0;
65      for (int i = 0; i <= 100; i++) {
66        for (int j = 0; j <= 100; j++) {
67          if (invertLoopIndexes) {
68            value = j;
69            value2 = i;
70          } else {
71            value = i;
72            value2 = j;
73          }
74          String[] split = callCalculateSuffix(value, value2, fullBlocks, partialBlocks);
75          for (String suffix : split) {
76            logger.trace(suffix);
77            String[] values = suffix.split("\\+", -1);
78            if (values.length > 1) {
79              try {
80                value = Integer.parseInt(values[0]);
81                value2 = Integer.parseInt(values[1]);
82              } catch (NumberFormatException e) {
83                Assertions.fail("NumberFormatException should never occur here");
84              }
85              // TODO JWL 12/12/2016 This never occurs so why do we care?
86              if (value > 5 || value2 > 5) {
87                count++;
88                StringBuilder msg = new StringBuilder();
89                msg.append("Found incorrect value ");
90                msg.append(suffix);
91                msg.append(". value = ");
92                msg.append(invertLoopIndexes ? j : i);
93                msg.append(" value2 = ");
94                msg.append(invertLoopIndexes ? i : j);
95                msg.append(" fullBlocks = ");
96                msg.append(fullBlocks);
97                msg.append(" partialBlocks = ");
98                msg.append(partialBlocks);
99                logger.trace(msg.toString());
100             }
101           }
102         }
103       }
104     }
105     Assertions.assertFalse(count > 0, "Incorrect values were founded " + count + " times");
106   }
107 
108   /**
109    * Call calculate suffix.
110    *
111    * @param value the value
112    * @param value2 the value2
113    * @param fullBlocks the full blocks
114    * @param partialBlocks the partial blocks
115    *
116    * @return the string[]
117    */
118   private static String[] callCalculateSuffix(int value, int value2, int fullBlocks,
119       int partialBlocks) {
120     String body = "{0} ";
121 
122     VisualScoreTag visualScoreTag = new VisualScoreTag();
123 
124     // Values are based on datasources_table.jsp
125     visualScoreTag.setFullBlocks(fullBlocks);
126     visualScoreTag.setPartialBlocks(partialBlocks);
127     visualScoreTag.setShowEmptyBlocks(true);
128     visualScoreTag.setShowA(true);
129     visualScoreTag.setShowB(true);
130 
131     visualScoreTag.setValue(value);
132     visualScoreTag.setValue2(value2);
133 
134     String buf = visualScoreTag.calculateSuffix(body);
135 
136     return buf.split(" ", -1);
137   }
