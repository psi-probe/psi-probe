19  class FlapListenerTests {

21    /** The default threshold. */
22    private final int defaultThreshold = 10;
23  
24    /** The default interval. */
25    private final int defaultInterval = 10;
26  
27    /** The default start threshold. */
28    private final float defaultStartThreshold = 0.29F;
29  
30    /** The default stop threshold. */
31    private final float defaultStopThreshold = 0.49F;
32  
33    /** The default low weight. */
34    private final float defaultLowWeight = 1.0F;
35  
36    /** The default high weight. */
37    private final float defaultHighWeight = 1.0F;
38  
39    /** The listener. */
40    private MockFlapListener listener = new MockFlapListener(defaultThreshold, defaultInterval,
41        defaultStartThreshold, defaultStopThreshold, defaultLowWeight, defaultHighWeight);
42  
43    /** The below threshold. */
44    private StatsCollectionEvent belowThreshold = new StatsCollectionEvent("test", 0, 0);
45  
46    /** The above threshold. */
47    private StatsCollectionEvent aboveThreshold = new StatsCollectionEvent("test", 0, 20);
48  
49    /**
50     * Fill.
51     *
52     * @param sce the sce
53     */
54    private void fill(StatsCollectionEvent sce) {
55      listener.reset();
56      add(sce, defaultInterval);
57    }
58  
59    /**
60     * Adds the.
61     *
62     * @param sce the sce
63     * @param quantity the quantity
64     */
65    private void add(StatsCollectionEvent sce, int quantity) {
66      for (int i = 0; i < quantity; i++) {
67        listener.statsCollected(sce);
68      }
69    }

71    /**
72     * Test below threshold not flapping.
73     */
74    @Test
75    void testBelowThresholdNotFlapping() {
76      listener.reset();
77      listener.statsCollected(aboveThreshold);
78      listener.statsCollected(belowThreshold);
79      Assertions.assertTrue(listener.isBelowThresholdNotFlapping());
80    }

90      Assertions.assertTrue(listener.isAboveThresholdNotFlapping());
91    }

199   /**
200    * The listener interface for receiving mockFlap events. The class that is interested in
201    * processing a mockFlap event implements this interface, and the object created with that class
202    * is registered with a component using the component's <code>addMockFlapListener</code> method.
203    * When the mockFlap event occurs, that object's appropriate method is invoked.
204    */
205   public static class MockFlapListener extends AbstractFlapListener {
206 
207     /** The threshold. */
208     private final long threshold;
209 
210     /** The flapping started. */
211     private boolean flappingStarted;
212 
213     /** The above threshold flapping stopped. */
214     private boolean aboveThresholdFlappingStopped;
215 
216     /** The below threshold flapping stopped. */
217     private boolean belowThresholdFlappingStopped;
218 
219     /** The above threshold not flapping. */
220     private boolean aboveThresholdNotFlapping;
221 
222     /** The below threshold not flapping. */
223     private boolean belowThresholdNotFlapping;
224 
225     /**
226      * Instantiates a new mock flap listener.
227      *
228      * @param threshold the threshold
229      * @param flapInterval the flap interval
230      * @param flapStartThreshold the flap start threshold
231      * @param flapStopThreshold the flap stop threshold
232      * @param lowWeight the low weight
233      * @param highWeight the high weight
234      */
235     public MockFlapListener(long threshold, int flapInterval, float flapStartThreshold,
236         float flapStopThreshold, float lowWeight, float highWeight) {
237 
238       this.threshold = threshold;
239       setDefaultFlapInterval(flapInterval);
240       setDefaultFlapStartThreshold(flapStartThreshold);
241       setDefaultFlapStopThreshold(flapStopThreshold);
242       setDefaultFlapLowWeight(lowWeight);
243       setDefaultFlapHighWeight(highWeight);
244     }
245 
246     @Override
247     public void statsCollected(StatsCollectionEvent sce) {
248       resetFlags();
249       super.statsCollected(sce);
250     }
251 
252     @Override
253     protected void flappingStarted(StatsCollectionEvent sce) {
254       flappingStarted = true;
255     }
256 
257     @Override
258     protected void aboveThresholdFlappingStopped(StatsCollectionEvent sce) {
259       aboveThresholdFlappingStopped = true;
260     }
261 
262     @Override
263     protected void belowThresholdFlappingStopped(StatsCollectionEvent sce) {
264       belowThresholdFlappingStopped = true;
265     }
266 
267     @Override
268     protected void aboveThresholdNotFlapping(StatsCollectionEvent sce) {
269       aboveThresholdNotFlapping = true;
270     }
271 
272     @Override
273     protected void belowThresholdNotFlapping(StatsCollectionEvent sce) {
274       belowThresholdNotFlapping = true;
275     }
276 
277     @Override
278     public long getThreshold(String name) {
279       return threshold;
280     }
281 
282     @Override
283     public void reset() {
284       resetFlags();
285       super.reset();
286     }
287 
288     /**
289      * Reset flags.
290      */
291     public void resetFlags() {
292       flappingStarted = false;
293       aboveThresholdFlappingStopped = false;
294       belowThresholdFlappingStopped = false;
295       aboveThresholdNotFlapping = false;
296       belowThresholdNotFlapping = false;
297     }
298 
299     /**
300      * Checks if is above threshold flapping stopped.
301      *
302      * @return true, if is above threshold flapping stopped
303      */
304     public boolean isAboveThresholdFlappingStopped() {
305       return aboveThresholdFlappingStopped;
306     }
307 
308     /**
309      * Checks if is above threshold not flapping.
310      *
311      * @return true, if is above threshold not flapping
312      */
313     public boolean isAboveThresholdNotFlapping() {
314       return aboveThresholdNotFlapping;
315     }
316 
317     /**
318      * Checks if is below threshold flapping stopped.
319      *
320      * @return true, if is below threshold flapping stopped
321      */
322     public boolean isBelowThresholdFlappingStopped() {
323       return belowThresholdFlappingStopped;
324     }
325 
326     /**
327      * Checks if is below threshold not flapping.
328      *
329      * @return true, if is below threshold not flapping
330      */
331     public boolean isBelowThresholdNotFlapping() {
332       return belowThresholdNotFlapping;
333     }
334 
335     /**
336      * Checks if is flapping started.
337      *
338      * @return true, if is flapping started
339      */
340     public boolean isFlappingStarted() {
341       return flappingStarted;
342     }
343 
344   }
